import threading
import time

# Global variables to store the current position and entry price
current_positions = {}
entry_prices = {}
action_taken = False

# Lock to synchronize access to shared variables
lock = threading.Lock()

# Function to continuously monitor the price and trigger the exit
def price_monitor(token, entry_price, profit_target):
    global action_taken
    while True:
        current_price = get_current_price(token)  # Replace with actual function to get current price
        if current_price is not None:
            profit = current_price - entry_price
            if profit >= profit_target:
                with lock:
                    if current_positions.get(token) is not None:
                        # Trigger the exit
                        entry_instrument = entry_instruments.get(token, set.ce_trading_token)
                        price = get_latest_price(entry_instrument)
                        if price is not None:
                            log_trade(time.time(), token, "PROFIT_EXIT", price)
                            print(f"Profit Exit Signal for {token} at {time.time()} at {pd.Timestamp.now()}")
                            current_positions[token] = None
                            action_taken = True
                break
        time.sleep(1)  # Adjust the sleep interval as needed

# Function to start the price monitoring thread
def start_price_monitoring(token, entry_price, profit_target):
    monitor_thread = threading.Thread(target=price_monitor, args=(token, entry_price, profit_target))
    monitor_thread.start()

# Example usage in process_token_data
def process_token_data(token):
    global action_taken
    df = resampled_data[token]    

    if len(df) > ema_length:
        # Extract the latest and previous EMA values
        EMA_short_latest = df['EMA_short'].iloc[-2]
        EMA_long_latest = df['EMA_long'].iloc[-2]
        ALMA_latest = df['ALMA'].iloc[-2]
        EMA_short_previous = df['EMA_short'].iloc[-3]
        EMA_long_previous = df['EMA_long'].iloc[-3]       

        timestamp = df.index[-1]
        
        with lock:
            current_position = current_positions.get(token)
            action_taken = False

        # Check for exit based on point-based profit target
        if current_position is not None:
            entry_price = entry_prices.get(token)
            current_price = df['close'].iloc[-1]
            profit = current_price - entry_price

            if (current_position == 'long' and profit >= 10) or (current_position == 'short' and profit <= -10):
                with lock:
                    if current_positions.get(token) is not None:
                        entry_instrument = entry_instruments.get(token, set.ce_trading_token if current_position == 'long' else set.pe_trading_token)
                        price = get_latest_price(entry_instrument)
                        if price is not None:
                            log_trade(timestamp, token, "PROFIT_EXIT", price)
                            print(f"Profit Exit Signal for {token} at {timestamp} at {pd.Timestamp.now()}")
                            current_positions[token] = None
                            action_taken = True

        # First, check for exit signals based on EMA cross
        if not action_taken:
            with lock:
                if current_position == 'long' and (EMA_short_latest < EMA_long_latest):
                    entry_instrument = entry_instruments.get(token, set.ce_trading_token)
                    price = get_latest_price(entry_instrument)
                    if price is not None:
                        log_trade(timestamp, token, "LONG_EXIT", price)
                        print(f"Long Exit Signal for {token} at {timestamp} at {pd.Timestamp.now()}")
                        current_positions[token] = None
                        action_taken = True

                elif current_position == 'short' and (EMA_short_latest > EMA_long_latest):
                    entry_instrument = entry_instruments.get(token, set.pe_trading_token)
                    price = get_latest_price(entry_instrument)
                    if price is not None:
                        log_trade(timestamp, token, "SHORT_EXIT", price)
                        print(f"Short Exit Signal for {token} at {timestamp} at {pd.Timestamp.now()}")
                        current_positions[token] = None
                        action_taken = True

        # Then, check for entry signals if we've just exited a position or if we're not in a position
        if action_taken or current_positions.get(token) is None:
            if (EMA_short_latest > EMA_long_latest >= ALMA_latest) and (EMA_short_previous <= EMA_long_previous):
                entry_instrument = set.ce_trading_token
                entry_instruments[token] = entry_instrument
                price = get_latest_price(entry_instrument)
                if price is not None:
                    log_trade(timestamp, token, "LONG_ENTRY", price)
                    print(f"Long Entry Signal for {token} at {timestamp} at {pd.Timestamp.now()}")
                    with lock:
                        current_positions[token] = 'long'
                        entry_prices[token] = price
                    start_price_monitoring(token, price, 10)  # Start monitoring for profit target

            elif (EMA_short_latest < EMA_long_latest <= ALMA_latest) and (EMA_short_previous >= EMA_long_previous):
                entry_instrument = set.pe_trading_token
                entry_instruments[token] = entry_instrument
                price = get_latest_price(entry_instrument)
                if price is not None:
                    log_trade(timestamp, token, "SHORT_ENTRY", price)
                    print(f"Short Entry Signal for {token} at {timestamp} at {pd.Timestamp.now()}")
                    with lock:
                        current_positions[token] = 'short'
                        entry_prices[token] = price
                    start_price_monitoring(token, price, -10)  # Start monitoring for profit target
